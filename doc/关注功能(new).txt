关注功能<------需要新增关注表（关注者ID,被关注者ID）

添加关注（已知两人ID）

insert into follow(followerID,followedID)
values(7,8)

取消关注(已知自己ID)

delete from follow
where (followedID = (select userID 
		from [user]
		where userName='zseqsc')
		and followerID = 8)

查看全部关注（已知自己ID）
select userName,userID
from [user],follow
where followedID = userID
and followerID = 3
